      <input id="playerInput" type="text" placeholder="Type your action..." />

      <button onclick="sendInput()">Enter</button>

  <script>

    let log = document.getElementById("log");

    let state = {};     // persistent memory

    let history = [];   // conversation memory


    async function sendInput() {

      const inputBox = document.getElementById("playerInput");

      const playerInput = inputBox.value.trim();

      if (!playerInput) return;

      inputBox.value = "";


      // Show player input in log

      log.textContent += `\n> ${playerInput}`;

      log.scrollTop = log.scrollHeight;


      // Save input in memory

      history.push({ role: "user", content: playerInput });


      try {

        const res = await fetch("/api/ai", {

          method: "POST",

          headers: { "Content-Type": "application/json" },

          body: JSON.stringify({ state, playerInput, history })

        });


        const data = await res.json();


        if (data.error) {

          log.textContent += `\n[Error: ${data.error}]`;

        } else {

          // Append AI reply

          log.textContent += `\n${data.narration}`;

          log.scrollTop = log.scrollHeight;


          // Update memory state

          if (data.state) state = data.state;


          // Save AI reply to history

          history.push({ role: "assistant", content: data.narration });

        }

      } catch (err) {

        log.textContent += `\n[Network error: ${err.message}]`;

      }

    }

  </script>
